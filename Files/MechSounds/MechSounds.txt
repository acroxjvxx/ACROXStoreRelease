<script>
    // --- SINTESI AUDIO MECCANICA AVANZATA ---
    let audioCtx = null;
    let selectedSwitchProfile = 'cherry-black';
    let mechVibesEnabled = true;
    let keySoundsEnabled = true;   
    let mouseSoundsEnabled = true; 
    
    // NUOVO: Moltiplicatori del volume (da 0 a 1)
    let keyVolume = 1.0;
    let mouseVolume = 1.0;

    function initAudioContext() {
        if (!audioCtx) {
            audioCtx = new (window.AudioContext || window.webkitAudioContext)();
        }
    }

    // Gestione cambi stato dei Toggle
    window.toggleMechVibes = function(enabled) {
        mechVibesEnabled = enabled;
    };

    window.toggleKeySounds = function(enabled) {
        keySoundsEnabled = enabled;
    };

    window.toggleMouseSounds = function(enabled) {
        mouseSoundsEnabled = enabled;
    };

    // NUOVO: Funzioni per aggiornare dinamicamente il volume
    window.updateKeyVolume = function(val) {
        keyVolume = parseFloat(val);
    };

    window.updateMouseVolume = function(val) {
        mouseVolume = parseFloat(val);
    };

    // Selettore dello switch
    window.selectSwitch = function(profile, element) {
        selectedSwitchProfile = profile;
        
        document.querySelectorAll('.switch-card').forEach(card => card.classList.remove('active'));
        element.classList.add('active');
        
        if (mechVibesEnabled && keySoundsEnabled) {
            playMechSound();
        }
    };

    // Sintetizza il suono meccanico applicando il volume della tastiera
    function playMechSound() {
        initAudioContext();
        if (audioCtx.state === 'suspended') {
            audioCtx.resume();
        }

        const now = audioCtx.currentTime;
        
        // Configurazione acustica degli switch
        let config = {
            noiseGainVal: 0.1,
            oscFreq: 120,
            oscGainVal: 0.25,
            decayTime: 0.08,
            clickFreq: 1200,
            clickGainVal: 0.0,
            clickDecay: 0.01
        };

        if (selectedSwitchProfile === 'cherry-black') {
            config.oscFreq = 85;
            config.oscGainVal = 0.45;
            config.noiseGainVal = 0.08;
            config.decayTime = 0.12;
            config.clickGainVal = 0.02;
            config.clickFreq = 300;
            config.clickDecay = 0.02;
        } else if (selectedSwitchProfile === 'cherry-blue') {
            config.oscFreq = 160;
            config.oscGainVal = 0.15;
            config.noiseGainVal = 0.18;
            config.decayTime = 0.05;
            config.clickGainVal = 0.45;
            config.clickFreq = 3200;
            config.clickDecay = 0.006;
        } else if (selectedSwitchProfile === 'cherry-red') {
            config.oscFreq = 110;
            config.oscGainVal = 0.3;
            config.noiseGainVal = 0.12;
            config.decayTime = 0.07;
            config.clickGainVal = 0.08;
            config.clickFreq = 1000;
            config.clickDecay = 0.01;
        } else if (selectedSwitchProfile === 'gateron-brown') {
            config.oscFreq = 135;
            config.oscGainVal = 0.35;
            config.noiseGainVal = 0.06;
            config.decayTime = 0.09;
            config.clickGainVal = 0.05;
            config.clickFreq = 700;
            config.clickDecay = 0.015;
        }

        // --- Generatore di Rumore ---
        const bufferSize = audioCtx.sampleRate * 0.2;
        const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
        const data = buffer.getChannelData(0);
        for (let i = 0; i < bufferSize; i++) {
            data[i] = Math.random() * 2 - 1;
        }

        const noiseNode = audioCtx.createBufferSource();
        noiseNode.buffer = buffer;

        const noiseFilter = audioCtx.createBiquadFilter();
        noiseFilter.type = 'bandpass';
        noiseFilter.frequency.value = config.oscFreq * 3;
        noiseFilter.Q.value = 1.0;

        const noiseGain = audioCtx.createGain();
        // NUOVO: Moltiplicato per keyVolume
        noiseGain.gain.setValueAtTime(config.noiseGainVal * keyVolume, now);
        noiseGain.gain.exponentialRampToValueAtTime(0.001, now + config.decayTime);

        noiseNode.connect(noiseFilter);
        noiseFilter.connect(noiseGain);
        noiseGain.connect(audioCtx.destination);

        // --- Generatore Tonale ---
        const osc = audioCtx.createOscillator();
        const oscGain = audioCtx.createGain();
        
        osc.type = 'triangle';
        osc.frequency.setValueAtTime(config.oscFreq, now);
        
        // NUOVO: Moltiplicato per keyVolume
        oscGain.gain.setValueAtTime(config.oscGainVal * keyVolume, now);
        oscGain.gain.exponentialRampToValueAtTime(0.001, now + config.decayTime);

        osc.connect(oscGain);
        oscGain.connect(audioCtx.destination);

        // --- Generatore Click ---
        if (config.clickGainVal > 0) {
            const clickOsc = audioCtx.createOscillator();
            const clickGain = audioCtx.createGain();
            
            clickOsc.type = 'sine';
            clickOsc.frequency.setValueAtTime(config.clickFreq, now);
            
            // NUOVO: Moltiplicato per keyVolume
            clickGain.gain.setValueAtTime(config.clickGainVal * keyVolume, now);
            clickGain.gain.exponentialRampToValueAtTime(0.001, now + config.clickDecay);
            
            clickOsc.connect(clickGain);
            clickGain.connect(audioCtx.destination);
            
            clickOsc.start(now);
            clickOsc.stop(now + config.clickDecay + 0.05);
        }

        noiseNode.start(now);
        noiseNode.stop(now + config.decayTime + 0.05);
        osc.start(now);
        osc.stop(now + config.decayTime + 0.05);
    }

    // --- SINTESI AUDIO MOUSE ---
    function playMouseSound(type) {
        initAudioContext();
        if (audioCtx.state === 'suspended') {
            audioCtx.resume();
        }

        const now = audioCtx.currentTime;

        let frequency = 2800;
        let gainValue = 0.3;
        let decay = 0.015;

        if (type === 'right') {
            frequency = 3200;
            gainValue = 0.25;
            decay = 0.012;
        } else if (type === 'wheel') {
            frequency = 1200;
            gainValue = 0.15;
            decay = 0.008;
        }

        const osc = audioCtx.createOscillator();
        const oscGain = audioCtx.createGain();
        osc.type = 'sine';
        osc.frequency.setValueAtTime(frequency, now);
        osc.frequency.exponentialRampToValueAtTime(100, now + decay);

        // NUOVO: Moltiplicato per mouseVolume
        oscGain.gain.setValueAtTime(gainValue * mouseVolume, now);
        oscGain.gain.exponentialRampToValueAtTime(0.0001, now + decay);

        osc.connect(oscGain);
        oscGain.connect(audioCtx.destination);

        const bufferSize = audioCtx.sampleRate * 0.05;
        const buffer = audioCtx.createBuffer(1, bufferSize, audioCtx.sampleRate);
        const data = buffer.getChannelData(0);
        for (let i = 0; i < bufferSize; i++) {
            data[i] = Math.random() * 2 - 1;
        }
        const noiseNode = audioCtx.createBufferSource();
        noiseNode.buffer = buffer;

        const noiseFilter = audioCtx.createBiquadFilter();
        noiseFilter.type = 'highpass';
        noiseFilter.frequency.value = type === 'wheel' ? 800 : 4000;

        const noiseGain = audioCtx.createGain();
        // NUOVO: Moltiplicato per mouseVolume
        noiseGain.gain.setValueAtTime(gainValue * 0.4 * mouseVolume, now);
        noiseGain.gain.exponentialRampToValueAtTime(0.0001, now + decay);

        noiseNode.connect(noiseFilter);
        noiseFilter.connect(noiseGain);
        noiseGain.connect(audioCtx.destination);

        osc.start(now);
        osc.stop(now + decay + 0.02);
        noiseNode.start(now);
        noiseNode.stop(now + decay + 0.02);
    }

    // --- INTERATTIVITÀ DELLA TASTIERA ---
    const activeKeys = new Set();

    document.querySelectorAll('.key').forEach(keyEl => {
        keyEl.addEventListener('mousedown', () => {
            keyEl.classList.add('active');
            if (mechVibesEnabled && keySoundsEnabled) playMechSound();
        });
        keyEl.addEventListener('mouseup', () => {
            keyEl.classList.remove('active');
        });
        keyEl.addEventListener('mouseleave', () => {
            keyEl.classList.remove('active');
        });
    });

    window.addEventListener('keydown', (e) => {
        if (document.activeElement.tagName === 'INPUT' || document.activeElement.tagName === 'TEXTAREA') {
            return;
        }

        let code = e.code;
        
        if (activeKeys.has(code)) {
            e.preventDefault();
            return;
        }
        activeKeys.add(code);

        const keyEl = document.querySelector(`.key[data-key="${code}"]`);
        if (keyEl) {
            e.preventDefault();
            keyEl.classList.add('active');
            if (mechVibesEnabled && keySoundsEnabled) playMechSound();
        }
    });

    window.addEventListener('keyup', (e) => {
        let code = e.code;
        activeKeys.delete(code);
        const keyEl = document.querySelector(`.key[data-key="${code}"]`);
        if (keyEl) {
            keyEl.classList.remove('active');
        }
    });

    // --- INTERATTIVITÀ DEL MOUSE ---
    const mouseLeft = document.getElementById('mouse-left');
    const mouseRight = document.getElementById('mouse-right');
    const mouseWheel = document.getElementById('mouse-wheel');

    document.addEventListener('contextmenu', (e) => {
        e.preventDefault();
    });

    document.addEventListener('mousedown', (e) => {
        if (e.button === 0) {
            mouseLeft.classList.add('active');
            if (mechVibesEnabled && mouseSoundsEnabled) playMouseSound('left');
        } else if (e.button === 2) {
            mouseRight.classList.add('active');
            if (mechVibesEnabled && mouseSoundsEnabled) playMouseSound('right');
        } else if (e.button === 1) {
            mouseWheel.classList.add('scrolling');
            if (mechVibesEnabled && mouseSoundsEnabled) playMouseSound('wheel');
        }
    });

    document.addEventListener('mouseup', (e) => {
        if (e.button === 0) {
            mouseLeft.classList.remove('active');
        } else if (e.button === 2) {
            mouseRight.classList.remove('active');
        } else if (e.button === 1) {
            mouseWheel.classList.remove('scrolling');
        }
    });

    document.addEventListener('wheel', () => {
        if (!mechVibesEnabled || !mouseSoundsEnabled) return;
        
        mouseWheel.classList.add('scrolling');
        playMouseSound('wheel');
        
        clearTimeout(window.wheelTimeout);
        window.wheelTimeout = setTimeout(() => {
            mouseWheel.classList.remove('scrolling');
        }, 120);
    }, { passive: true });
</script>
