<script>
    let cacheImmagineCaricata = null;
    let cacheTipoCaricamento = "";
    let inputTargetCorrente = null;
    let vecchioNomePerModifica = null;

    function apriModaleCaricamento(event, tipo) {
        const file = event.target.files[0];
        if (!file) return;
        inputTargetCorrente = event.target;
        
        // 🔍 CONTROLLO FORMATO FILE RIGIDO INTEGRATO
        const nomeFile = file.name.toLowerCase();
        const tipoFile = file.type.toLowerCase();
        const esitoGif = nomeFile.endsWith('.gif') || tipoFile === 'image/gif';

        if (tipo === 'free' && esitoGif) {
            if (typeof openAcroxAlert === "function") {
                openAcroxAlert("⚠️ Formato Non Supportato", "La libreria Standard (Senza AISOS) accetta solo immagini statiche (.png, .jpg, .jpeg). Le GIF animate sono una funzionalità Premium!", "error");
            } else {
                alert("⚠️ La libreria Standard accetta solo immagini statiche (.png, .jpg). Le GIF sono per utenti Premium.");
            }
            event.target.value = "";
            return;
        }

        // Controllo validità generale estensioni ammesse
        const formatiAccettati = ['image/png', 'image/jpeg', 'image/jpg', 'image/gif'];
        if (!formatiAccettati.includes(tipoFile) && !nomeFile.endsWith('.png') && !nomeFile.endsWith('.jpg') && !nomeFile.endsWith('.jpeg') && !nomeFile.endsWith('.gif')) {
            if (typeof openAcroxAlert === "function") {
                openAcroxAlert("⚠️ File Non Valido", "Puoi caricare solo file in formato .png, .jpg, .jpeg o .gif!", "error");
            } else {
                alert("⚠️ Puoi caricare solo file in formato .png, .jpg, .jpeg o .gif!");
            }
            event.target.value = "";
            return;
        }

        let emojiSalvate = JSON.parse(localStorage.getItem('acrox_emoji_list')) || [];
        const countFree = emojiSalvate.filter(e => e.tipo === 'free').length;
        const countPremium = emojiSalvate.filter(e => e.tipo === 'premium').length;

        if (tipo === 'free' && countFree >= 10) {
            if (typeof openAcroxAlert === "function") {
                openAcroxAlert("⚠️ Limite Raggiunto", "Hai raggiunto il limite massimo di 10 emoji per la libreria Standard (Senza AISOS)!", "error");
            } else {
                alert("⚠️ Limite di 10 emoji raggiunto per la libreria Standard!");
            }
            event.target.value = "";
            return;
        }
        if (tipo === 'premium' && countPremium >= 100) {
            if (typeof openAcroxAlert === "function") {
                openAcroxAlert("⚠️ Limite Raggiunto", "Hai raggiunto il limite massimo di 100 emoji per la libreria Premium (Con AISOS)!", "error");
            } else {
                alert("⚠️ Limite di 100 emoji raggiunto per la libreria Premium!");
            }
            event.target.value = "";
            return;
        }

        const reader = new FileReader();
        reader.onload = function(e) {
            cacheImmagineCaricata = e.target.result;
            cacheTipoCaricamento = tipo;
            vecchioNomePerModifica = null;

            document.getElementById('emoji-modal-preview').src = cacheImmagineCaricata;
            document.getElementById('emoji-custom-name').value = "";
            document.getElementById('emoji-name-modal').style.display = 'flex';
        };
        reader.readAsDataURL(file);
    }

    function apriModaleModificaNome(nomeAttuale, srcAttuale, tipoAttuale) {
        vecchioNomePerModifica = nomeAttuale;
        cacheImmagineCaricata = srcAttuale;
        cacheTipoCaricamento = tipoAttuale;

        const nomePulito = nomeAttuale.replace(/:/g, '');

        document.getElementById('emoji-modal-preview').src = srcAttuale;
        document.getElementById('emoji-custom-name').value = nomePulito;
        document.getElementById('emoji-name-modal').style.display = 'flex';
    }

    function chiudiModaleCaricamento() {
        document.getElementById('emoji-name-modal').style.display = 'none';
        cacheImmagineCaricata = null;
        cacheTipoCaricamento = "";
        vecchioNomePerModifica = null;
        if(inputTargetCorrente) inputTargetCorrente.value = "";
    }

    function confermaSalvataggioEmoji() {
        let nomeInput = document.getElementById('emoji-custom-name').value.trim();
        if (nomeInput === "") {
            if (typeof openAcroxAlert === "function") {
                openAcroxAlert("⚠️ Errore", "Il nome dell'emoji non può essere vuoto!", "error");
            } else {
                alert("⚠️ Il nome dell'emoji non può essere vuoto!");
            }
            return;
        }

        nomeInput = nomeInput.toLowerCase().replace(/:/g, '');
        const nomeFormattato = `:${nomeInput}:`;
        let emojiSalvate = JSON.parse(localStorage.getItem('acrox_emoji_list')) || [];

        const esisteGia = emojiSalvate.some(e => e.nome === nomeFormattato && e.nome !== vecchioNomePerModifica);
        if (esisteGia) {
            if (typeof openAcroxAlert === "function") {
                openAcroxAlert("⚠️ Errore", `Il nome "${nomeFormattato}" è già in uso! Scegli un altro nome.`, "error");
            } else {
                alert(`⚠️ Il nome "${nomeFormattato}" è già in uso!`);
            }
            return;
        }

        if (vecchioNomePerModifica) {
            const index = emojiSalvate.findIndex(e => e.nome === vecchioNomePerModifica);
            if (index !== -1) {
                emojiSalvate[index].nome = nomeFormattato;
            }
        } else {
            emojiSalvate.push({
                nome: nomeFormattato,
                src: cacheImmagineCaricata,
                tipo: cacheTipoCaricamento
            });
        }

        localStorage.setItem('acrox_emoji_list', JSON.stringify(emojiSalvate));
        
        document.getElementById('emoji-name-modal').style.display = 'none';
        vecchioNomePerModifica = null;
        if(inputTargetCorrente) inputTargetCorrente.value = "";

        if (typeof caricaLibrerieSalvate === "function") {
            caricaLibrerieSalvate();
        }
    }
</script>
